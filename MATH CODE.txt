    <!-- App Logic Script -->
    <script>
        // 1. Math Equations Infrastructure 
        function calculateWeight(x) {
            // f(x) = 3200 / (1 + e^(-0.15 * (x - 24)))
            return 3200 / (1 + Math.exp(-0.15 * (x - 24)));
        }

        function calculateDerivative(x) {
            // f'(x) = (480 * e^(-0.15 * (x - 24))) / (1 + e^(-0.15 * (x - 24)))^2
            const expTerm = Math.exp(-0.15 * (x - 24));
            return (480 * expTerm) / Math.pow(1 + expTerm, 2);
        }

        // 2. Generate Dataset for the production cycle (Day 0 to Day 45)
        const labels = [];
        const weightData = [];
        const derivativeData = [];

        for (let day = 0; day <= 45; day++) {
            labels.push(`Day ${day}`);
            weightData.push(calculateWeight(day).toFixed(1));
            derivativeData.push(calculateDerivative(day).toFixed(1));
        }

        // 3. Dynamic Layman Insights Generation engine based on derivative values
        function updateLaymanInterpretation(day, weight, rate) {
            let message = "";
            const numDay = parseInt(day);

            if (numDay < 14) {
                message = `The chicks are currently in the starter phase. Total weight is low (${weight}g), but the growth derivative is climbing steadily from ${rate}g/day. Focus on maintaining brooding temperatures and high-protein feed initialization.`;
            } else if (numDay >= 14 && numDay < 23) {
                message = `The birds are in their grower phase. The derivative is accelerating dramatically (${rate}g/day). Skeletal structures are rapidly widening to accommodate heavy muscle tissue. Ensure clean ventilation in the broiler shed.`;
            } else if (numDay >= 23 && numDay <= 25) {
                message = `Biological Peak Performance attained! Growth rate has topped out at ${rate}g/day. The birds are transforming feed into breast meat at maximum absolute efficiency. Your investment in nutrition yields its highest daily return right now.`;
            } else if (numDay > 25 && numDay <= 38) {
                message = `The broilers are heavily muscled. While cumulative weight continues to reach a massive ${weight}g, the growth speed derivative has dropped back down to ${rate}g/day. Metabolic maintenance requirements are starting to compete with weight gain.`;
            } else {
                message = `Market maturity reached. Total mass is stable near ${weight}g, but growth velocity has plummeted drastically down to ${rate}g/day. Keeping them past this window increases feed costs without generating profitable weight gains. Time for harvest processing.`;
            }

            document.getElementById('out-day').innerText = `Day ${day}`;
            document.getElementById('out-weight').innerText = `${weight} g`;
            document.getElementById('out-rate').innerText = `${rate} g/day`;
            document.getElementById('out-layman').innerText = message;
            document.getElementById('active-tag').innerText = "Data Extracted";
            document.getElementById('active-tag').classList.remove('bg-teal-500/20', 'text-teal-300');
            document.getElementById('active-tag').classList.add('bg-emerald-500/20', 'text-emerald-300');
        }

        // 4. Initialize ChartJS with smooth configurations
        const ctx = document.getElementById('growthChart').bind('context') ? document.getElementById('growthChart') : document.getElementById('growthChart').getContext('2d');
        
        const growthChart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: labels,
                datasets: [
                    {
                        label: 'Total Weight f(x) (grams)',
                        data: weightData,
                        borderColor: '#10b981',
                        backgroundColor: 'rgba(16, 185, 129, 0.05)',
                        borderWidth: 3,
                        yAxisID: 'yWeight',
                        tension: 0.2,
                        pointRadius: 1,
                        pointHoverRadius: 6,
                        pointHitRadius: 15
                    },
                    {
                        label: "Growth Velocity f'(x) (g/day)",
                        data: derivativeData,
                        borderColor: '#14b8a6',
                        backgroundColor: 'transparent',
                        borderWidth: 3,
                        borderDash: [5, 5],
                        yAxisID: 'yRate',
                        tension: 0.2,
                        pointRadius: 1,
                        pointHoverRadius: 6,
                        pointHitRadius: 15
                    }
                ]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                animation: {
                    duration: 1800,
                    easing: 'easeOutQuart'
                },
                scales: {
                    x: {
                        grid: { display: false },
                        ticks: { color: '#64748b', font: { size: 11 } }
                    },
                    yWeight: {
                        type: 'linear',
                        position: 'left',
                        title: { display: true, text: 'Cumulative Weight (grams)', color: '#10b981', font: { weight: 'bold' } },
                        min: 0,
                        max: 3500,
                        ticks: { color: '#64748b' }
                    },
                    yRate: {
                        type: 'linear',
                        position: 'right',
                        title: { display: true, text: 'Daily Rate of Change (g/day)', color: '#14b8a6', font: { weight: 'bold' } },
                        min: 0,
                        max: 140,
                        grid: { drawOnChartArea: false },
                        ticks: { color: '#64748b' }
                    }
                },
                plugins: {
                    legend: { display: false }
                },
                // Intercept clicks on graph elements to update calculus information
                onClick: (e) => {
                    const activePoints = growthChart.getElementsAtEventForMode(e, 'nearest', { intersect: false }, true);
                    if (activePoints.length > 0) {
                        const index = activePoints[0].index;
                        const dayNumber = index; // index tracks day 0 to 45 perfectly
                        const selectedWeight = weightData[dayNumber];
                        const selectedRate = derivativeData[dayNumber];
                        
                        updateLaymanInterpretation(dayNumber, selectedWeight, selectedRate);
                    }
                }
            }
        });
    </script>
